\section*{Appendix: Working-Memory-Friendly Lesson Flow Figures}

\textbf{Design note:} These visual figures are meant to support teachers and students who benefit from seeing the whole learning path at once. Each figure separates the build sequence, teacher support, and evidence so the lesson can be followed without holding every step in working memory.

\begin{figure}[htbp]
\centering
\begin{tikzpicture}[
stepbox/.style={rounded corners=3pt, draw=LessonOneAccent, fill=LessonOneSoft, thick, align=center, minimum height=0.48in, text width=1.05in, font=\scriptsize},
widebox/.style={rounded corners=3pt, draw=LessonOneAccent, fill=white, thick, align=left, text width=6.65in, inner sep=5pt, font=\scriptsize},
arrow/.style={-{Stealth[length=2mm]}, thick, draw=LessonOneAccent}
]
\node[stepbox] (a) {Notice\\meaning};
\node[stepbox, right=0.13in of a] (b) {Choose\\place};
\node[stepbox, right=0.13in of b] (c) {Sketch\\features};
\node[stepbox, right=0.13in of c] (d) {Begin\\prototype};
\node[stepbox, right=0.13in of d] (e) {Explain\\choice};
\draw[arrow] (a) -- (b);
\draw[arrow] (b) -- (c);
\draw[arrow] (c) -- (d);
\draw[arrow] (d) -- (e);
\node[widebox, below=0.18in of c] (support) {\textbf{Teacher support:} Model decomposition, protect private identity choices, offer roles, and show the first build step.};
\node[widebox, below=0.10in of support] (evidence) {\textbf{Evidence:} Planning card, prototype or sketch, partner explanation, and exit slip.};
\end{tikzpicture}
\caption{Lesson 1 visual flow: Plan a Meaningful Place.}
\end{figure}
